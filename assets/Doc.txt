-- Table des disciplines
CREATE TABLE discipline (
    id_discipline INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(100) NOT NULL
);

-- Table des utilisateur (clients ou coachs)
CREATE TABLE utilisateur (
    id_utilisateur INT AUTO_INCREMENT PRIMARY KEY,
    nom_complet VARCHAR(100) NOT NULL,
    pseudo VARCHAR(100) UNIQUE NOT NULL,
    date_inscription DATE NOT NULL,
    type_abonnement VARCHAR(100) NOT NULL,
    mot_de_passe VARCHAR(150) NOT NULL,
    photo VARCHAR(150),
    role ENUM('client', 'coach') NOT NULL,
    id_discipline INT,
    FOREIGN KEY (id_discipline) REFERENCES discipline(id_discipline)
);

-- Table de planification des coachs par semaine
CREATE TABLE affectation_coach (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_coach INT NOT NULL,
    semaine DATE NOT NULL,
    FOREIGN KEY (id_coach) REFERENCES utilisateurs(id)
);

-- Table des équipements
CREATE TABLE equipement (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nom VARCHAR(100) NOT NULL,
    quantite INT NOT NULL,
    etat ENUM ("bon", "usé", "en réparation") NOT NULL
);
